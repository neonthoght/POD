-- 1) Совпадает марка только на английском и известна модель
-- 13131 из 41294 строк
select e.model, g.name_brand, g.code_brand, g.code_model, g.name_model
from pod.errors e
join pod.guide g on 
	e.model ilike '%' || g.code_brand || '%'
	and e.normal not ilike '%' || g.name_brand || '%'
	and e.normal ilike '%' || g.name_model || '%';


-- 2) Совпадает марка на русском и известна модель
select e.model, g.name_brand, g.code_brand, g.code_model, g.name_model
from pod.errors e
join pod.guide g on 
	and e.normal not ilike '%' || g.name_brand || '%'
	and e.normal ilike '%' || g.name_model || '%';
